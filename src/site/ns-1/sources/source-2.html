123 	 * The parsing method fixes some input date mistakes, i.e. june 31th will be
124 	 * parsed to july, 1st. February 29th of a non leap year will be parsed to
125 	 * March, 1st.
126 	 * @param inputDate <code>String</code> with the date to to convert.
127 	 * @param inputFormat <code>String</code> with the input format. A built-in format

129 	 * @param outputFormat <code>String</code> with the output format. A built-in format
130 	 *                      can be used or a user's custom one.
131 	 * @throws ParseException when the input date is not parseable because doesn't match the input format.
132 	 * @return <code>String</code> converted to the output format.
133 	 * @since v2.3.6
134 	 * @see SimpleDateFormat
135 	 * @see ParseException
136 	 */
137 	public static String dateFormatChanger(String inputDate, String inputFormat, String outputFormat) throws ParseException {
138 		//Defines input default format.
139 		String input = inputFormat.isEmpty() ? "yyyyMMdd" : inputFormat;
140 		//Defines output default format.
141 		String output = outputFormat.isEmpty() ? "yyyyMMdd" : outputFormat;
142 		String outputDate;
143 		outputDate = new SimpleDateFormat(output, LOCALE_ES).format(new SimpleDateFormat(input, LOCALE_ES).parse(inputDate));
144 		return outputDate;
145 	}

178 		long endEpochDay = endExclusive.toEpochDay();
